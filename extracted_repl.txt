        // 2. Fetch Live News & Generate Summary
        async function fetchReliefWebNews() {
            const feedContainer = document.getElementById('news-feed');
            try {
                // Gunakan URL API yang lebih sederhana
                const url = 'https://api.reliefweb.int/v1/reports?appname=gprc&query[value]=Gaza&sort[]=date:desc&limit=10';
                const response = await fetch(url);
                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }
                const data = await response.json();

                if (data && data.data && data.data.length > 0) {
                    let html = '';
                    data.data.forEach(item => {
                        // Handle date if it exists
                        let dateStr = 'Terkini';
                        if (item.fields && item.fields.date && item.fields.date.created) {
                            dateStr = new Date(item.fields.date.created).toLocaleDateString('id-ID', {
                                day: 'numeric', month: 'short', year: 'numeric'
                            });
                        }
                        const title = item.fields ? item.fields.title : item.title;
                        const link = item.fields && item.fields.url ? item.fields.url : item.url || '#';
                        html += `
                            <div class="news-item">
                                <div class="news-date">${dateStr}</div>
                                <div class="news-title">
                                    <a href="${link}" target="_blank">${title}</a>
                                </div>
                            </div>
                        `;
                    });
                    feedContainer.innerHTML = html;
                    generateAutoSummary(data.data); // Generate the relevant summary
                } else {
                    feedContainer.innerHTML = '<div style="text-align:center; padding:20px; color:red;">Data berita tidak ditemukan.</div>';
                }
            } catch (error) {
                console.error("Error fetching news:", error);
                feedContainer.innerHTML = `<div style="text-align:center; padding:20px; color:red;">Koneksi API gagal: ${error.message}</div>`;
            }
        }

        // 3. Auto Summary Logic
        function generateAutoSummary(articles) {
            let kwHealth = 0, kwFood = 0, kwShelter = 0;
            articles.forEach(item => {
                const titleRaw = item.fields && item.fields.title ? item.fields.title : item.title || "";
                const title = titleRaw.toLowerCase();
                if(title.includes('health') || title.includes('hospital') || title.includes('medical') || title.includes('disease')) kwHealth++;
                if(title.includes('food') || title.includes('hunger') || title.includes('famine') || title.includes('water')) kwFood++;
                if(title.includes('shelter') || title.includes('displaced') || title.includes('refugee') || title.includes('camp') || title.includes('evacuation')) kwShelter++;
            });

            let analysis = `Berdasarkan 10 laporan kemanusiaan aktual terbaru dari PBB, `;
            let issues = [];
            
            if (kwHealth > 0) issues.push('krisis kesehatan dan kolapsnya fasilitas medis');
            if (kwFood > 0) issues.push('ancaman kelaparan (famine) dan krisis air bersih');
            if (kwShelter > 0) issues.push('gelombang pengungsian masif (IDPs) akibat evakuasi');

            if (issues.length > 0) {
                analysis += `isu yang sangat mendesak dan mendominasi saat ini berpusat pada <strong>${issues.join(', ')}</strong>. `;
                analysis += `Fakta di lapangan ini menegaskan argumen utama riset Anda: Intervensi dari otoritas formal (BoP/NCAG) tidak akan cukup cepat jika dikendalikan dari luar. Tata kelola rekonstruksi yang ideal (model AMTG) wajib mengintegrasikan <em>Local Ownership</em>, di mana komite-komite lokal diberdayakan untuk menjadi garda terdepan dalam mendistribusikan pasokan darurat dan mengelola area pengungsian yang dinamis.`;
            } else {
                analysis += `terdapat volatilitas krisis yang berkelanjutan di lapangan. Model AMTG yang diusulkan sangat relevan untuk menyediakan tata kelola transisi yang adaptif dan merangkul legitimasi akar rumput demi merespons dinamika ini secara lincah dan inklusif.`;
            }

            document.getElementById('auto-summary-text').innerHTML = analysis;
            document.getElementById('auto-summary-box').style.display = 'block';
        }
